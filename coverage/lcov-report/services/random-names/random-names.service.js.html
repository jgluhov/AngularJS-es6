export default class RandomNames {
  constructor() {
    this.names = ['John', 'Elisa', 'Mark', 'Annie'];
  }
 
  getName() {
    const totalNames = this.names.length;
    const randomIndex = Math.floor(Math.random() * totalNames);
    return this.names[randomIndex];
  }
}
